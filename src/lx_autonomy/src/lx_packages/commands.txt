    "{parameters: [{name: \"autodig.pid_outer\", value: {type: 8, double_array_value: [0.015, 0.000001, 0.02]}},\
    {name: \"autodig.pid_inner\", value: {type: 8, double_array_value: [20, 0.0001, 0.5]}}]}"
